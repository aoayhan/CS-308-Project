npm install express
npm install firebase-admin


add song:
curl -X POST http://localhost:3000/api/add-song \
-H "Content-Type: application/json" \
-d '{"songName": "Psycho", "album": "PSYCHODRAMA", "artist": "Dave", "year": 2019, "rating": 8, "userId": "asd@mail.com"}'

add json batch: 
curl -X POST http://localhost:3000/api/add-batch-songs \
-H "Content-Type: application/json" \
--data-binary "@songs.json"

delete by artist:
curl -X DELETE http://localhost:3000/api/delete-songs-by-artist \
-H "Content-Type: application/json" \
-d '{"artist": "Sample+Artist"}'

search in spoti:
curl -X GET "http://localhost:3000/spotify-search?q=Nirvana"

search in database:
curl -X GET "http://localhost:3000/api/search-song?name=anil"

export artist:
curl -X GET "http://localhost:3000/api/export-songs?artist=Jorja%20Smith" -o songs.json
